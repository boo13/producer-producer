"""Sentence-level classification of job descriptions for visual tiering."""

from __future__ import annotations

import re


def split_into_sentences(text: str | None) -> list[str]:
    """Split text into sentences, handling job description formatting.

    Splits on:
    - Period/exclamation/question mark followed by space and uppercase letter
    - Double newlines (paragraph boundaries)
    - Bullet points (â€˘, -, *) at start of lines

    Preserves abbreviations like U.S., e.g., etc.
    """
    if not text:
        return []

    # Normalize whitespace within lines but preserve paragraph breaks
    text = text.strip()

    # Split on paragraph boundaries first
    paragraphs = re.split(r"\n\s*\n", text)

    sentences = []
    for para in paragraphs:
        para = para.strip()
        if not para:
            continue

        # Check for bullet-point lists
        lines = para.split("\n")
        if len(lines) > 1 and all(
            re.match(r"^\s*[â€˘\-\*â–Şâ–ş]\s", line) or re.match(r"^\s*\d+[.)]\s", line) or line.strip() == ""
            for line in lines[1:]
            if line.strip()
        ):
            # It's a list â€” treat each line as a sentence
            for line in lines:
                line = line.strip()
                if line:
                    sentences.append(line)
            continue

        # Sentence-split within the paragraph
        # Use regex that splits on sentence-ending punctuation followed by space
        # but not after common abbreviations
        parts = re.split(r"(?<=[.!?])\s+(?=[A-Z])", para)
        for part in parts:
            part = " ".join(part.split()).strip()
            if part:
                sentences.append(part)

    return sentences
```

import json
import logging

logger = logging.getLogger(__name__)

VALID_TIERS = {"highlight", "normal", "faded", "very_faded"}

CLASSIFICATION_SYSTEM_PROMPT = """You are a job posting analyst. Classify each sentence into one of four visual tiers:

- "highlight": Experience requirements, salary/compensation details, must-have qualifications, key role details that a job seeker needs to see immediately.
- "normal": Role description, team structure, day-to-day responsibilities, reporting lines â€” useful information about the job itself.
- "faded": Generic company description, benefits boilerplate, perks lists â€” nice to know but not essential for evaluating the role.
- "very_faded": Legal disclaimers, EEO statements, compliance language, standard legal boilerplate.

Respond with a JSON array. Each element must have "text" (the exact sentence) and "tier" (one of: highlight, normal, faded, very_faded).

Example:
[
  {"text": "5+ years of experience required.", "tier": "highlight"},
  {"text": "You will work with a talented team.", "tier": "normal"},
  {"text": "We offer great benefits.", "tier": "faded"},
  {"text": "We are an equal opportunity employer.", "tier": "very_faded"}
]"""


def classify_sentences(
    sentences: list[str],
    provider,
    config: dict,
) -> list[dict] | None:
    """Classify sentences into visual tiers using LLM.

    Args:
        sentences: List of sentence strings to classify.
        provider: LLM provider with .generate() method.
        config: AI config dict (temperature, max_tokens, etc.).

    Returns:
        List of {"text": str, "tier": str} dicts, or None on error.
    """
    if not sentences:
        return []

    user_prompt = "Classify these sentences:\n\n" + json.dumps(sentences)

    try:
        response = provider.generate(
            system_prompt=CLASSIFICATION_SYSTEM_PROMPT,
            user_prompt=user_prompt,
            temperature=config.get("temperature", 0.0),
            max_tokens=config.get("max_tokens", 4096),
        )
    except Exception:
        logger.exception("Sentence classification failed")
        return None

    # Parse response â€” provider may return parsed JSON or raw string
    classifications = response if isinstance(response, list) else _parse_response(response)
    if classifications is None:
        return None

    # Validate and sanitize tiers
    result = []
    for item in classifications:
        if not isinstance(item, dict) or "text" not in item:
            continue
        tier = item.get("tier", "normal")
        if tier not in VALID_TIERS:
            tier = "normal"
        result.append({"text": item["text"], "tier": tier})

    return result


def _parse_response(response) -> list[dict] | None:
    """Parse LLM response into list of classification dicts."""
    if isinstance(response, str):
        try:
            parsed = json.loads(response)
            if isinstance(parsed, list):
                return parsed
        except json.JSONDecodeError:
            logger.error("Failed to parse classification response as JSON")
            return None
    if isinstance(response, dict):
        # Some providers wrap in a content key
        content = response.get("content") or response.get("choices", [{}])[0].get("message", {}).get("content")
        if content:
            return _parse_response(content)
    return None
```
